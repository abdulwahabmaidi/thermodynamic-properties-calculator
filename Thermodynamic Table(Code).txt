    /* Variables for user input and choice*/
    double Temp, Pres;
    int choice;
    
    /* Loop for menu display and user interaction*/
    do {
        printf("\nEnter (1) or (2), T-P Table or exit: ");
        scanf("%d", &choice);
        
        switch (choice) {
            /* Case for displaying T-P Table*/
            case 1:
                printf("Temperature\tPressure\n");
                for (int i = 0; i < 21; ++i) {
                    printf("%.2f\t\t%.4f\n", Temperature[i], Pressure[i]);
                }
                
                /* Ask user for values calculation*/
                int entropyChoice;
                printf("Enter 2 to find values of water: ");
                scanf("%d", &entropyChoice);

                    /* Loop for values calculation*/
                    do {
                        printf("Enter the value of T and P: ");
                        scanf("%lf %lf", &Temp, &Pres);
                        int foundTemp = 0;
                        int foundPres = 0;
                        
                        /* Check if the T and P values are in the table*/
                        for (int i = 0; i < 21; i++) {
                            if (Temp == Temperature[i] && Pres == Pressure[i]) {
                                foundTemp = 1;
                                foundPres = 1;
                                double x;
                                printf("It is Saturated, so enter the value of x (must be 0-1): ");
                                scanf("%lf", &x);
                                double h = hf[i] + x * hfg[i];
                                double v = vf[i] + x * vfg[i];
                                double s = sf[i] + x * sfg[i];
                                printf("values for x = %.2f is:\n enthalpy = %.4f kj/kg\n specific volume =%.4f m^3/kg\n entropy = %.4f kj/kg*kmol \n ", x, h,v,s);
                                break;
                            }
                        }
                        
                        /* If T and P values are not found in the table*/
                        if (!foundTemp || !foundPres) {
                            int compressed = 0;
                            int superheated = 0;
                            /* Check if the state is compressed or superheated*/
                            for (int i = 0; i < 21; i++) {
                                if ((Temp == Temperature[i] && Pres > Pressure[i]) ||
                                    (Temp < Temperature[i] && Pres == Pressure[i])) {
                                    compressed = 1;
                                    printf("It is Compressed \n *Note that values can be determined using interpolation"); 
                                }
                                if ((Temp == Temperature[i] && Pres < Pressure[i]) ||
                                    (Temp > Temperature[i] && Pres == Pressure[i])) {
                                    superheated = 1;
                                    printf("It is Superheated\n *Note that values can be determined using interpolation\n");
                                }
                            }
                            if (!compressed && !superheated) {
                                printf("Values not found in the table.\n"); 
                            }
                        }
                        
                        /* Ask user if they want to continue calculating values*/
                        int continueEntropy;
                        printf("Do you want to continue calculating values of water? (1 for yes, 0 for no): ");
                        scanf("%d", &continueEntropy);
                        if (continueEntropy == 0) {
                            break;}
                    } while (1); }
                break;
            case 2:
                printf("Exiting program.\n");
                return 0;
            default:
                printf("Invalid choice. Please enter 1 or 2.\n");
                break;
        }
    } while (1);
    return 0;
}
